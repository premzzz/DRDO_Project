#include <stdio.h>
#include <stdlib.h>

// Global or static file pointer accessible to the cleanup function
static FILE *_cov_f = NULL;

// This function runs automatically ONLY on a successful/perfect exit
void close_coverage_file(void) {
    if (_cov_f) {
        fclose(_cov_f); // Safely flushes and saves the final data
    }
}

#define LOG_FLAG(name) do {                           \
    if (!_cov_f) {                                    \
        _cov_f = fopen("coverage.log", "a");          \
        if (_cov_f) atexit(close_coverage_file);      \
    }                                                 \
    if (_cov_f) fprintf(_cov_f, "%s\n", name);        \
} while(0)



## FLUSING MEMORY TYPE 2

COVERAGE_H = '''\
/* coverage.h – auto-generated by instrument_project.py */
#ifndef COVERAGE_H
#define COVERAGE_H
 
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
 
/*
 * LOG_FLAG(name)
 * --------------
 * Records one hit of an execution-block flag.
 *
 * Performance note: writing to disk on every single hit (fopen/fprintf/
 * fclose per call) is extremely slow in tight loops that run thousands or
 * millions of times. Instead, hits are accumulated in a small in-memory
 * buffer and flushed to coverage.log in large batches:
 *   - The buffer fills up to COV_BUFFER_HITS hits, then is written to disk
 *     in ONE fwrite() call and cleared.
 *   - The file is opened once (kept open for the program's lifetime) and
 *     only closed at exit, avoiding repeated fopen/fclose overhead.
 *   - An atexit() handler flushes any remaining buffered hits when the
 *     program ends normally, so no data is lost on a clean exit.
 *
 * Output format is UNCHANGED — still one flag name per line in
 * coverage.log — so existing tooling (coverage_report.py, html_report.py)
 * needs no changes.
 */
 
#define COV_BUFFER_HITS   1000      /* hits per batch before flushing   */
#define COV_LINE_MAXLEN   64        /* max chars per flag name + newline */
 
static FILE *_cov_file = NULL;
static char  _cov_buffer[COV_BUFFER_HITS][COV_LINE_MAXLEN];
static int   _cov_count = 0;
 
static void _cov_flush(void) {
    if (_cov_count == 0) return;
 
    if (!_cov_file) {
        _cov_file = fopen("coverage.log", "a");
    }
    if (_cov_file) {
        for (int _i = 0; _i < _cov_count; _i++) {
            fputs(_cov_buffer[_i], _cov_file);
        }
        fflush(_cov_file);
    }
    _cov_count = 0;
}
 
static void _cov_close(void) {
    _cov_flush();
    if (_cov_file) {
        fclose(_cov_file);
        _cov_file = NULL;
    }
}
 
static int _cov_atexit_registered = 0;
 
#define LOG_FLAG(name) do {                                    \\
    if (!_cov_atexit_registered) {                             \\
        atexit(_cov_close);                                    \\
        _cov_atexit_registered = 1;                            \\
    }                                                           \\
    snprintf(_cov_buffer[_cov_count], COV_LINE_MAXLEN, "%s\\n", name); \\
    _cov_count++;                                               \\
    if (_cov_count >= COV_BUFFER_HITS) {                        \\
        _cov_flush();                                           \\
    }                                                            \\
} while(0)
 
#endif /* COVERAGE_H */
'''
